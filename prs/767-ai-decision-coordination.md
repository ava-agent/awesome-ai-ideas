class SimpleResourceScheduler:

        self.allocation_rules = {}
        
    def simple_allocation(self, task_requirements, available_resources):
        # 基于规则的简单分配
        allocation = {}
        for resource_type, required in task_requirements.items():
            available = available_resources.get(resource_type, 0)
            if available >= required:
                allocation[resource_type] = required
            else:
                allocation[resource_type] = available
        return allocation
    
    def basic_demand_prediction(self, historical_data, time_period):
